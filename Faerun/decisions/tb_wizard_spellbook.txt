#Small explanation as for how wizard spellcasting works:

decisions = {
	
	comprehend_languages = {
		ai_check_interval = 12
		potential = {
			OR = {
				AND = {
					ai = yes
					OR = {
						has_job_title = job_chancellor
						is_ruler = yes
					}
				}
				has_character_flag = wiz_spellbook_open
			}
		}
		allow = {
			custom_tooltip = {
				text = HASSPELLSLOTS
				hidden_tooltip = {
					OR = {
						AND = {
							OR = {
								trait = wizard
								trait = trained_wizard
							}
							NOT = { check_variable = { wizminor = 3 } }
						}
						AND = {
							OR = {
								trait = journeyman_wizard
								trait = expert_wizard
								trait = master_wizard
								trait = renowned_wizard
								trait = legendary_wizard
							}
							NOT = { change_variable = { wizminor = 4 } }
						}
					}
				}
			}
		}
		effect = {
			add_character_modifier = {
				name = comprehend_languages_mod
				months = 3
			}
			hidden_tooltip = { change_variable = { which = wizminor value = 1 } }
		}
		ai_will_do = {
			factor = 0.1
			modifier = {
				factor = 0.1
				diplomacy = 10 # Don't really need it
			}
			modifier = {
				factor = 5
				NOT = { diplomacy = 5 }
			}
		}
	}
	
}

targeted_decisions = {
	
	cantrip_friends = { # This will be a player-only spell because the AI can't really deal with the strategy
		filter = realm
		only_playable = yes
		
		from_potential = {
			z_wizard = yes
			ai = no
		}
		
		potential = {
			always = yes
		}
		
		allow = {
			custom_tooltip = {
				text = CANTRIP_ON_COOLDOWN
				hidden_tooltip = {
					NOT = {
						had_character_flag = {
							flag = spell_friends_cooldown
							months = 1
						}
					}
				}
			}
			NOT = {
				has_opinion_modifier = {
					who = FROM
					name = friends_spell
				}
			}
			NOT = {
				has_opinion_modifier = {
					who = FROM
					name = friends_spell_after
				}
			}
		}
		
		effect = {
			spell_resistance_check = yes
			opinion = {
				modifier = friends_spell
				who = FROM
				months = 3
			}
			repeat_event = { id = spellaux.2 days = 90 }
		}
	}
	
}
